def self.init_with_program(prog)
  prog.command(:share) do |c|
    c.syntax 'share'
    c.description 'Share latest post with each configured backend'
    c.option 'dry', '-d', '--dry', 'perform dry run'
    c.action { |args, opts| action(args, opts) }
  end
end
